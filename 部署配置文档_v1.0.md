   324|## 四、双向数据同步（核心配置）
   325|
   326|### 4.1 同步脚本
   327|
   328|文件位置：`/root/.hermes/scripts/campus-community-sync.sh`
   329|（同时复制到 `/root/backup-site/sync.sh`）
   330|
   331|```bash
   332|#!/bin/bash
   333|# ============================================
   334|# 校园社区 - 双向数据同步脚本
   335|# 主站 ←→ 备份站，数据完全一致
   336|# 使用 sqlite3 .backup 确保一致性快照
   337|# ============================================
   338|set -e
   339|
   340|BACKUP_DIR="/root/backup-site"
   341|LOG_FILE="/var/log/campus-sync.log"
   342|SSH_CMD="ssh -p 28022"
   343|
   344|log() {
   345|    echo "[$(date '+%Y-%m-%d %H:%M:%S')] $1" | tee -a "$LOG_FILE"
   346|}
   347|
   348|log "========== 开始双向同步 =========="
   349|
   350|# 1. 创建一致快照（不停服，.backup 保证一致性）
   351|log "创建主站 DB 快照..."
   352|$SSH_CMD root@104.168.15.71 \
   353|    "sqlite3 /root/campus-community/prisma/dev.db '.backup /tmp/dev.db.snapshot'"
   354|
   355|log "创建备份站 DB 快照..."
   356|sqlite3 "$BACKUP_DIR/prisma/dev.db" '.backup /tmp/dev.db.backup-snapshot'
   357|
   358|# 2. 传输主站快照到本机
   359|log "[主→备] 传输 DB 快照..."
   360|rsync -avz -e "$SSH_CMD" \
   361|    root@104.168.15.71:/tmp/dev.db.snapshot \
   362|    /tmp/dev.db.primary-snapshot
   363|
   364|# 3. 比较时间戳，选最新的
   365|PRI_MTIME=$(stat -c %Y /tmp/dev.db.primary-snapshot 2>/dev/null || echo 0)
   366|BAK_MTIME=$(stat -c %Y /tmp/dev.db.backup-snapshot 2>/dev/null || echo 0)
   367|
   368|if [ "$PRI_MTIME" -ge "$BAK_MTIME" ]; then
   369|    log "→ 主站数据较新"
   370|    WINNER="/tmp/dev.db.primary-snapshot"
   371|else
   372|    log "→ 备份站数据较新"
   373|    WINNER="/tmp/dev.db.backup-snapshot"
   374|fi
   375|
   376|# 4. 用选中版本更新两台服务器
   377|# 4a. 更新备份站
   378|log "更新备份站..."
   379|pm2 stop campus-community
   380|cp "$WINNER" "$BACKUP_DIR/prisma/dev.db"
   381|pm2 restart campus-community
   382|
   383|# 4b. 更新主站
   384|log "更新主站..."
   385|rsync -avz -e "$SSH_CMD" \
   386|    "$WINNER" \
   387|    root@104.168.15.71:/tmp/dev.db.final
   388|
   389|$SSH_CMD root@104.168.15.71 \
   390|    "systemctl stop campus-life && \
   391|     cp /tmp/dev.db.final /root/campus-community/prisma/dev.db && \
   392|     systemctl restart campus-life"
   393|
   394|# 5. 上传文件双向同步
   395|log "[主→备] 同步上传文件..."
   396|rsync -avz --update --delete -e "$SSH_CMD" \
   397|    root@104.168.15.71:/root/campus-community/public/uploads/ \
   398|    "$BACKUP_DIR/public/uploads/"
   399|
   400|log "[备→主] 同步上传文件..."
   401|rsync -avz --update --delete -e "$SSH_CMD" \
   402|    "$BACKUP_DIR/public/uploads/" \
   403|    root@104.168.15.71:/root/campus-community/public/uploads/
   404|
   405|# 6. 清理临时文件
   406|rm -f /tmp/dev.db.snapshot /tmp/dev.db.primary-snapshot \
   407|      /tmp/dev.db.backup-snapshot /tmp/dev.db.final
   408|$SSH_CMD root@104.168.15.71 \
   409|    "rm -f /tmp/dev.db.snapshot /tmp/dev.db.backup-snapshot /tmp/dev.db.final" || true
   410|
   411|log "========== 双向同步完成 =========="
   412|```
